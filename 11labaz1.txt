#include <stdio.h>
#include <math.h>

#define N 10 // Определение размера массива

int main() {
    float A[N]; // Объявление массива вещественных чисел
    float transformed[N]; // Массив для хранения преобразованных значений
    float temp; // Временная переменная для ввода
    float sum = 0; // Для вычисления суммы
    float avg; // Среднее

    // Заполнение массива значениями
    for(int i = 0; i < N; i++) {
        printf("a[%d] = ", i + 1);
        scanf("%f", &temp);
        A[i] = temp; // Заполняем массив
    }

    // Преобразование элементов массива
    for(int i = 0; i < N; i++) {
        transformed[i] = A[i] * 2; // Удвоить значение каждого элемента массива
    }

    // Вывод результатов
    printf("\n| индекс | исходное значение | новое значение |\n");
    for (int i = 0; i < N; i++) {
        printf("|   %d   |      %.2f       |     %.2f      |\n", i + 1, A[i], transformed[i]);
    }

    // Вычисление среднего арифметического
    for (int i = 0; i < N; i++) {
        sum += A[i]; // Суммируем значения исходного массива
    }
    avg = sum / N; // Получаем среднее
    printf("Среднее арифметическое: %.2f\n", avg);

    return 0; // Завершение программы
}
