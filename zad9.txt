#include <stdio.h>
#include <math.h>

int main() {
    float h, a, angle;
    char fill_char; // Символ для рисования

    // Ввод параметров
    printf("Введите высоту параллелограмма (h): ");
    scanf("%f", &h);
    printf("Введите основание параллелограмма (a): ");
    scanf("%f", &a);
    printf("Введите угол наклона (в градусах): ");
    scanf("%f", &angle);
    printf("Введите символ для заполнения: ");
    scanf(" %c", &fill_char); // Обратите внимание на пробел перед %c для игнорирования пробелов

    // Перевод угла из градусов в радианы
    angle = angle * (M_PI / 180); // Преобразование в радианы
    
    // Вычисляем величины для отображения
    int space_offset = (int)(a * cos(angle)); // Сдвиг для начальных пробелов
    int height = (int)h; // Целочисленное значение высоты

    // Рисование параллелограмма
    for (int i = 0; i < height; i++) {
        // Печать пробелов для смещения
        for (int j = 0; j < space_offset; j++) {
            printf(" ");
        }

        // Печать заполненной строки параллелограмма
        for (int j = 0; j < (int)a; j++) {
            printf("%c", fill_char);
        }

        // Переход на следующую строку
        printf("\n");
    }

    return 0; // Завершение программы
}
