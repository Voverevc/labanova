#include <stdio.h>
#include <math.h> 
#include <locale.h>// Для использования функции cos

int main() {
    setlocale(LC_ALL, "RUS");
    int n; // Переменная для хранения значения n
    double x; // Переменная для хранения значения x
    double sum = 0.0; // Переменная для накопления суммы результатов

    // Ввод значений n и x
    printf("Введите натуральное число n: ");
    scanf("%d", &n);
    printf("Введите действительное число x: ");
    scanf("%lf", &x);

    // Проверка, что n натуральное
    if (n <= 0) {
        printf("Ошибка: n должно быть натуральным числом.\n");
        return 1;
    }

    // Вычисление суммы
    for (int i = 1; i <= n; i++) {
        sum += cos(pow(x, i)); // Вычисление cos(x^i) и добавление к сумме
    }

    // Вывод результата
    printf("Сумма cos(x) + cos(x^2) + ... + cos(x^%d) = %lf\n", n, sum);

    return 0; // Завершение программы
}
